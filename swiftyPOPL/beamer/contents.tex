\section{Motivation}

\begin{frame}{Question: What Is the Typed Meaning?}
\small
Swift-style protocols are convenient surface syntax, but their typed core
meaning is less direct.

\vspace{2mm}
\begin{center}
\setlength{\tabcolsep}{4pt}
\renewcommand{\arraystretch}{1.08}
\begin{tabular}{lll}
\toprule
Surface feature & Static role & Core idea \\
\midrule
\texttt{protocol P} & required methods & dictionary type \\
\texttt{extension S: P} & concrete conformance & dictionary value \\
\texttt{T: P} & generic capability assumption & dictionary parameter \\
\texttt{any P} & hidden conforming value & existential package \\
\bottomrule
\end{tabular}
\end{center}

\vspace{2mm}
\begin{block}{Goal}
Design a small Swift-like language whose protocol features elaborate to a
standard typed core calculus.
\end{block}

\[
  \text{well-typed surface program}
  \quad\Longrightarrow\quad
  \text{well-typed elaborated core program}
\]
\end{frame}

\section{Surface Typing}

\begin{frame}[fragile]{Running Example}
\begin{lstlisting}[
  style=swiftypopl,
  basicstyle=\ttfamily\scriptsize,
  aboveskip=1mm,
  belowskip=0pt
]
protocol ReusableCell {
  func render() -> String
}

struct FeedCell { 
  let id: String 
  func getId() -> String { id }
}

extension FeedCell: ReusableCell {
  func render() -> String { self.getId() }
}

let keep: <T: ReusableCell>(cell: T) -> T =
  func <T: ReusableCell>(cell: T) -> T {
    if true { cell } else { cell }
  }

let cell: FeedCell = FeedCell(id: "feed")

let erased: any ReusableCell = cell as any ReusableCell
\end{lstlisting}
\end{frame}

\begin{frame}{Surface Typing Grammar}
\small
\[
\begin{array}{rcll}
\tau
&::=&
\Int \mid \Bool \mid \String \mid \Unit
& \text{base types}
\\
&\mid&
S \mid X \mid \Self
& \text{struct, type variable, contextual self}
\\
&\mid&
\Any{P}
& \text{existential protocol value}
\\
&\mid&
\ftype{\overline{X:\overline{P}}}
       {\overline{\ell:\tau}}
       {\tau}
& \text{first-class function scheme}
\\[1mm]
\ell
&::=&
\epsilon \mid \kw{label}
& \text{positional or labeled parameter}
\end{array}
\]
\end{frame}

\begin{frame}{Typing Contexts}
\small
\[
\Psi=(\Psi_S,\Psi_P,\Psi_C,\Psi_M,\Psi_V)
\]

\begin{center}
\scriptsize
\setlength{\tabcolsep}{3pt}
\renewcommand{\arraystretch}{1.05}
\begin{tabular}{lll}
\toprule
Context & Stores & Example \\
\midrule
$\Psi_S(S)$ & struct fields & $\Psi_S(\kw{FeedCell})=\{\kw{id}:\String\}$ \\
$\Psi_P(P)$ & protocol requirements & $\kw{render}:\Self\to\String$ \\
$\Psi_C(S,P)$ & conformances & $\kw{FeedCell}:\kw{ReusableCell}$ \\
$\Psi_M(S)$ & methods of a type & $\kw{FeedCell.render}$ \\
$\Psi_V(x)$ & top-level values & $\kw{keep}:\forall T.\ldots$ \\
\bottomrule
\end{tabular}
\end{center}

\vspace{1mm}
\begin{itemize}
  \item $\Omega$ records generic constraints such as $T:\kw{ReusableCell}$.
  \item $\Gamma$ records local values and parameters.
  \item $\rho$ resolves \Self: concrete in structs/extensions, abstract in protocols.
\end{itemize}
\end{frame}

\begin{frame}{Checker: Deciding Well-Typedness}
\scriptsize
\[
\inferrule*[Right=\textsc{T-TypeApp}]
  {
    \Psi;\Omega;\Gamma;\rho \vdash e:
      \ftype{\overline{X_i:\overline{P_i}}}
             {\overline{\ell_j:\tau_j}}
             {\tau_r}
    \\
    \overline{
      \Psi;\Omega \vdash \tau_i' \modelsP P
      \quad\text{for every }P\in\overline{P_i}}
  }
  {
    \Psi;\Omega;\Gamma;\rho \vdash
      e\langle\overline{\tau_i'}\rangle:
      \subst{\overline{\tau_i'}}{\overline{X_i}}
      \left(\ftype{\emptyctx}{\overline{\ell_j:\tau_j}}{\tau_r}\right)
  }
\]

\vspace{-1mm}
\[
\inferrule*[Right=\textsc{T-Call}]
  {
    \Psi;\Omega;\Gamma;\rho \vdash e :
      \ftype{\emptyctx}{\overline{\ell_i:\tau_i}}{\tau}
    \\
    \kw{matchArgs}(\overline{a_i},\overline{\ell_i:\tau_i})
      = \overline{\hat{a_i}:\tau_i}
  }
  {
    \Psi;\Omega;\Gamma;\rho \vdash e(\overline{a_i}) : \tau
  }
\]

\vspace{-2mm}
\begin{center}
\textsc{T-ConstrainedMethod}

\resizebox{0.92\textwidth}{!}{$
\inferrule*
  {
    \Psi;\Omega;\Gamma;\rho \vdash e:X
    \\
    X:\overline{P}\in\Omega
    \\
    P_k\in\overline{P}
    \\
    \Psi_P(P_k)(m)=
      \ftype{\overline{Y_i:\overline{Q_i}}}
             {\epsilon:\Self,\overline{\ell_j:\tau_j}}
             {\tau_r}
  }
  {
    \Psi;\Omega;\Gamma;\rho \vdash e.m:
      [\Self\mapsto X]
      \left(
        \ftype{\overline{Y_i:\overline{Q_i}}}
               {\overline{\ell_j:\tau_j}}
               {\tau_r}
      \right)
  }
$}
\end{center}
\end{frame}

\section{Elaboration}


\begin{frame}{Type Translation}
\small
\begin{block}{Target calculus}
System F with records, existentials, conditionals, recursion, and natural
numbers.
\end{block}
\small
\[
\begin{array}{rcl}
\trans{\Int} &=& \Nat \\
\trans{\Bool} &=& \Bool \\
\trans{\String} &=& \String \\
\trans{\Unit} &=& \Unit \\
\trans{S} &=& \{\overline{x:\trans{\tau}}\}
  \quad \text{where }\Psi_S(S)=\{\overline{x:\tau}\}
\end{array}
\]

\end{frame}

\begin{frame}{Conformance Translation}
\[
\Psi_P(P)=\{\overline{m_i:\sigma_i}\}
\qquad
\Dict(P,\trans{S})
=
\{\overline{m_i:\transSelf{\trans{S}}{\sigma_i}}\}
\]

\vspace{1mm}
\[
\inferrule*[]
  {
    (S,P)\in\dom(\Psi_C)
    \\
    \overline{
      \mu(S,m_i):
      \transSelf{\trans{S}}{\sigma_i}}
    \\
    \delta(S,P)
    =
    \{\overline{m_i=\mu(S,m_i)}\}
  }
  {
    \kw{extension}\ S:P\ \{\overline{m_i}\}
    \elab
    \delta(S,P):\Dict(P,\trans{S})
  }
\]

\vspace{1mm}
\begin{block}{Checker obligation}
Every requirement $m_i:\sigma_i$ of $P$ is implemented by $S$ after
substituting protocol \Self\ with $S$.
\end{block}
\end{frame}

\begin{frame}{Generic Constraints Become Dictionary Parameters}
\small
For a constrained generic function:
\[
\kw{func}\ f\langle T:P\rangle(x:T)\to R\ \{e\}
\]

the generated core has the shape:
\[
\begin{aligned}
f
&=
\Lambda T.\
\lambda d_{T,P}:\Dict(P,T).\
\lambda x:T.\ \trans{e}
\end{aligned}
\]

\vspace{1mm}
At an explicit type application, evidence is passed immediately:
\[
\begin{aligned}
f\langle S\rangle(v)
&\elab
f\ [\trans{S}]\ \delta(S,P)\ \trans{v}
\end{aligned}
\]

\vspace{-1mm}
Constrained method calls use the dictionary carried by $\Delta$:
\[
\inferrule*[]
  {
    \Psi;\Omega;\Gamma;\rho;\Delta \vdash e:X \elab t
    \\
    X:\overline{P}\in\Omega
    \quad
    P_k\in\overline{P}
    \quad
    \Delta(X,P_k)=d
    \\
    \Psi_P(P_k)(m)=
      \ftype{\emptyctx}{\epsilon:\Self,\overline{\ell_i:\tau_i}}{\tau_r}
    \\
    \overline{a}\elab\overline{u}
  }
  {
    e.m(\overline{a})
    \elab
    d.m\ t\ \overline{u}
  }
\]
\end{frame}

\begin{frame}{\texorpdfstring{\Any{P}}{any P} Is an Existential Package}
\small
\[
\trans{\Any{P}}
=
\exists X.\{\kw{value}:X,\kw{dict}:\Dict(P,X)\}
\]

\vspace{1mm}
Packaging a concrete value:
\[
\begin{aligned}
&\Psi;\Omega;\Gamma;\rho;\Delta \vdash e:\tau\elab t
\qquad
\Psi;\Omega;\Delta \vdash \tau\modelsP P\elab d
\\[-0.5mm]
&\Rightarrow\quad
e\ \kw{as}\ \Any{P}
\elab
\kw{pack}[\trans{\tau},\{\kw{value}=t,\kw{dict}=d\}]
\ \kw{as}\ \trans{\Any{P}}
\end{aligned}
\]

\vspace{1mm}
Calling an existential-safe method:
\[
\begin{aligned}
&\Psi;\Omega;\Gamma;\rho;\Delta \vdash e:\Any{P}\elab t
\qquad
\overline{a}\elab\overline{u}
\\[-0.5mm]
&\Rightarrow\quad
e.m(\overline{a})
\elab
\kw{unpack}\ \{X,z\}=t\ \kw{in}\
z.\kw{dict}.m\ z.\kw{value}\ \overline{u}
\end{aligned}
\]
\end{frame}

\section{Safety}

\begin{frame}{Preservation Theorem}
\small
\begin{block}{Elaboration preservation}
If a surface expression type checks and elaborates to a core term, then the
core term has the translated type.
\end{block}

\[
\Psi;\Omega;\Gamma;\rho\vdash e:\tau
\qquad
\Psi;\Omega;\Gamma;\rho;\Delta\vdash e:\tau\elab t
\]
\[
\Longrightarrow
\qquad
\mathcal{E}\vdash_c t:\trans{\tau}
\]

\[
  \mathcal{E}
  =
  \mathcal{G}_\Psi;\trans{\Omega};\trans{\Gamma};\trans{\Delta}
\]

\vspace{1mm}
\begin{itemize}
  \item $\mathcal{G}_\Psi$ types global values, method functions, and dictionaries.
  \item $\Delta$ types local dictionary evidence for active generic constraints.
  \item Core soundness then gives end-to-end safety.
\end{itemize}
\end{frame}

\begin{frame}{Proof Sketch: The Hard Cases}
\small
\begin{center}
\scriptsize
\setlength{\tabcolsep}{4pt}
\renewcommand{\arraystretch}{1.12}
\begin{tabular}{p{0.28\textwidth}p{0.62\textwidth}}
\toprule

\midrule

\midrule

\midrule

\midrule

\midrule
Declarations &
methods and conformances inhabit the global signature $\mathcal{G}_\Psi$ \\
\bottomrule
\end{tabular}
\end{center}
\end{frame}

\section{Conclusion}

\begin{frame}{Takeaways}
\small
\begin{itemize}
  \item \lang\ isolates the typed core of Swift-style protocols.
  \item Protocol conformance is explicit dictionary evidence.
  \item Generic constraints become dictionary parameters.
  \item \Any{P} is a standard existential package.
  \item The preservation proof connects surface typing to core typing.
\end{itemize}

\vspace{2mm}
\begin{block}{Summary}
Swift-like protocol features can be safely elaborated into System F.
\end{block}
\end{frame}

\begin{frame}{Backup: Protocol Satisfaction Produces Evidence}
\small
Only two surface facts can produce protocol evidence.

\vspace{-1mm}
\[
\begin{gathered}
\inferrule*[Right=\textsc{Sat-Struct}]
  {(S,P)\in\dom(\Psi_C)}
  {\Psi;\Omega\vdash S\modelsP P}
\\[2mm]
\inferrule*[Right=\textsc{Sat-TVar}]
  {X:\overline{P}\in\Omega \\ P\in\overline{P}}
  {\Psi;\Omega\vdash X\modelsP P}
\end{gathered}
\]

\vspace{2mm}
\begin{itemize}
  \item A concrete struct satisfies a protocol through an extension.
  \item A type variable satisfies a protocol through a generic constraint.
  \item \Any{P} does \emph{not} itself satisfy $P$.
\end{itemize}

\end{frame}

\begin{frame}{Backup: Core Type Translation}
\small
\[
\begin{array}{rcl}
\trans{\Int} &=& \Nat \\
\trans{\Bool} &=& \Bool \\
\trans{\String} &=& \String \\
\trans{\Unit} &=& \Unit \\
\trans{S} &=& \{\overline{x:\trans{\tau}}\}
  \quad \text{where }\Psi_S(S)=\{\overline{x:\tau}\} \\
\trans{\Any{P}} &=&
  \exists X.\{\kw{value}:X,\kw{dict}:\Dict(P,X)\}
\end{array}
\]

\[
\Dict(P,A)
=
\{\overline{m_i:\transSelf{A}{\sigma_i}}\}
\quad\text{where }\Psi_P(P)=\{\overline{m_i:\sigma_i}\}
\]

\vspace{1mm}
\begin{itemize}
  \item Protocol requirements use a self-parameterized translation.
  \item In concrete conformances, protocol \Self\ is instantiated with the struct type.
\end{itemize}
\end{frame}

\begin{frame}{Backup: Existential Method Rule}

  \Lambda\overline{Y_i}.\
